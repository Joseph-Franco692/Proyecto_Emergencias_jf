import { Component, OnInit } from '@angular/core';
import { CommonModule } from '@angular/common'; // <-- ESTO ARREGLA EL *ngIf, *ngFor Y EL PIPE date
import { WebsocketService } from './services/websocket';

@Component({
  selector: 'app-root',
  standalone: true,
  imports: [CommonModule], // <-- REGISTRAMOS EL MODULO DE DIRECTIVAS COMUNES AQUÍ
  templateUrl: './app.html',
  styleUrls: ['./app.css']
})
export class AppComponent implements OnInit { // <-- CLASE COMPONENTE PRINCIPAL
  title = 'Central Operativa de Bomberos';
  public listaReportes: any[] = [];

  // Inyectamos el servicio sin usar new() gracias a Angular
  constructor(private wsService: WebsocketService) {}

  ngOnInit(): void {
    // Escuchamos el flujo del WebSocket en tiempo real
    this.wsService.escucharNuevosReportes().subscribe({
      next: (reporteRecibido) => {
        console.log('🚨 ¡Nuevo reporte capturado en tiempo real!', reporteRecibido);
        this.listaReportes.unshift(reporteRecibido); // Lo añade al inicio de la lista
      },
      error: (err) => console.error('Error al escuchar el flujo de datos:', err)
    });
  }
}

import { Injectable } from '@angular/core';
import { Client } from '@stomp/stompjs';
import { Observable, Subject } from 'rxjs';

@Injectable({
  providedIn: 'root' // Hace que el servicio sea un componente global (Singleton). Angular maneja su ciclo de vida y evita el uso de new() en los componentes.
})
export class WebsocketService {
  private stompClient!: Client;
  
  // Un Subject funciona como un emisor y receptor de eventos intermedio. 
  // Nos servirá para redirigir los datos que llegan del servidor hacia cualquier pantalla de Angular.
  private reportesSubject: Subject<any> = new Subject<any>();

  constructor() {
    this.configurarWebSocket();
  }

  private configurarWebSocket() {
    // Inicializamos el cliente de STOMP apuntando a la dirección exacta de tu servidor de Spring Boot (Puerto 8081)
    this.stompClient = new Client({
      brokerURL: 'ws://localhost:8081/ws-emergencias', // El endpoint que expusimos en Java
      connectHeaders: {},
      debug: (str) => console.log('STOMP Debug Log:', str), // Imprime en la consola del navegador los latidos y estados de la conexión
      reconnectDelay: 5000, // Intento de reconexión automática cada 5 segundos si el servidor llega a caerse
      heartbeatIncoming: 4000, // Validaciones de vida de conexión cada 4 segundos
      heartbeatOutgoing: 4000
    });

    // Evento que se ejecuta automáticamente cuando la conexión se establece con éxito
    this.stompClient.onConnect = (frame) => {
      console.log('--- ¡CONECTADO CON ÉXITO AL WEBSOCKET DE BOMBEROS! ---');
      
      // NOS SUSCRIBIMOS al canal de transmisión de reportes creado en Spring Boot
      this.stompClient.subscribe('/topic/nuevos-reportes', (message) => {
        if (message.body) {
          // Parseamos el string JSON que nos envía Spring Boot y lo transformamos en un objeto nativo de JavaScript
          const nuevoReporte = JSON.parse(message.body);
          
          // Enviamos el objeto al flujo del Subject para que las pantallas se enteren
          this.reportesSubject.next(nuevoReporte);
        }
      });
    };

    // Control de errores en el protocolo en tiempo real
    this.stompClient.onStompError = (frame) => {
      console.error('Error detectado en el Broker STOMP: ', frame.headers['message']);
    };

    // Activamos formalmente la conexión de la tubería de datos
    this.stompClient.activate();
  }

  /**
   * Este método expone el flujo del Subject como un Observable.
   * Permitirá que las pantallas (Componentes) se queden "escuchando" indefinidamente las alertas entrantes.
   */
  public escucharNuevosReportes(): Observable<any> {
    return this.reportesSubject.asObservable();
  }
}
